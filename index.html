                def get_changed_dirs(git_path: Path, from_commit_hash: str, to_commit_hash: str) -> set[Path]:
                    command = f'git -C {git_path} diff --dirstat=files,0 {from_commit_hash} {to_commit_hash}'
                    output = subprocess.check_output(command, shell=True).decode('utf-8')
                    print(output)
                    changed_directories = [Path(git_path, line.split()[1]) for line in output.splitlines()]
                    return set(changed_directories)
